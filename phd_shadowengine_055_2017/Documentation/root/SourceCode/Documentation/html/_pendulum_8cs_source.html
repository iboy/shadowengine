128  Vector3 PendulumUpdate(Vector3 currentStatePosition, float deltaTime)
129  {
130  // Add gravity free fall
131  this.gravityForce = this.mass * Physics.gravity.magnitude;
132  this.gravityDirection = Physics.gravity.normalized;
133  this.currentVelocity += this.gravityDirection * this.gravityForce * deltaTime;
134 
135  Vector3 pivot_p = this.Pivot.transform.position;
136  Vector3 bob_p = this.currentStatePosition;
137 
138 
139  Vector3 auxiliaryMovementDelta = this.currentVelocity * deltaTime;
140  float distanceAfterGravity = Vector3.Distance(pivot_p, bob_p + auxiliaryMovementDelta);
141 
142  // If at the end of the rope
143  if(distanceAfterGravity > this.ropeLength || Mathf.Approximately(distanceAfterGravity, this.ropeLength))
144  {
145 
146  this.tensionDirection = (pivot_p - bob_p).normalized;
147 
148  this.pendulumSideDirection = (Quaternion.Euler(0f, 90f, 0f) * this.tensionDirection);
149  this.pendulumSideDirection.Scale(new Vector3(1f, 0f, 1f));
150  this.pendulumSideDirection.Normalize();
151 
152  this.tangentDirection = (-1f * Vector3.Cross(this.tensionDirection, this.pendulumSideDirection)).normalized;
153 
154 
155  float inclinationAngle = Vector3.Angle(bob_p-pivot_p, this.gravityDirection);
156 
157  this.tensionForce = this.mass * Physics.gravity.magnitude * Mathf.Cos(Mathf.Deg2Rad * inclinationAngle);
158  float centripetalForce = ((this.mass * Mathf.Pow(this.currentVelocity.magnitude, 2))/this.ropeLength);
159  this.tensionForce += centripetalForce;
160 
161  this.currentVelocity += this.tensionDirection * this.tensionForce * deltaTime;
162  }
163 
164  // Get the movement delta
165  Vector3 movementDelta = Vector3.zero;
166  movementDelta += this.currentVelocity * deltaTime;
167 
168 
169  //return currentStatePosition + movementDelta;
170 
171  float distance = Vector3.Distance(pivot_p, currentStatePosition + movementDelta);
172  return this.GetPointOnLine(pivot_p, currentStatePosition + movementDelta, distance <= this.ropeLength ? distance : this.ropeLength);
173  }
